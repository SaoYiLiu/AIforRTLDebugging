## Rationale
- **Bug #1 (lines 77–79):** In MSB-first mode with parity enabled, the load path assigns `shift_reg_d[8:4] = {p_data_i[4:0], parity_bit}` and `shift_reg_d[3:0] = p_data_i[8:5]`, which swaps the high and low nibbles instead of loading `p_data_i` unchanged.
- **Harness evidence for Bug #1:** For `BIT_ORDER=1`, `PARITY=1`, `p_data_i=8'hA5`, the harness expects `shift_reg_d = 9'b101001010` but the buggy RTL produces `9'b010110010` — exactly the nibble-scrambled value cited in the problem statement.
- **Bug #1 fix:** MSB-first with parity should append parity as the LSB of the shift register: `shift_reg_d = {p_data_i, parity_bit}`. This mirrors the LSB-first path (`{parity_bit, p_data_i}`) with parity placement reversed per spec.
- **Bug #2 (lines 99–105):** In `ST_TX` with `BIT_ORDER=1` and `EXTRA_BIT=1`, the FSM shifts left by **2** bits per cycle and increments `count_d` by 2, skipping serialized bits and misaligning output.
- **Harness evidence for Bug #2:** With correct load value `101001010`, expected serial output is `1,0,1,0,0,1,0,1,0` but the actual (from the waveform analysis) is `0,1,0,1,1,0,0,1,0` — consistent with 2-bit shifts corrupting the MSB-first stream. The cocotb timeout on `test_data_serializer[1-1-8]` and `[2-1-8]` occurs because `test_basic_transmission` never sees the expected bit sequence.
- **Bug #2 fix:** Use the same 1-bit left shift for all MSB-first transmissions, regardless of parity: `shift_reg_d = {shift_reg_q[SHIFT_W-2:0], 1'b0}; count_d = count_q + 1;`
- **Verification against failing cases:** `8'hA5` with even parity → `9'b101001010` → serial `101001010`; `8'h3C` with odd parity (XOR=0, inverted=1) → `9'b001111000` → serial `001111000`. Both match the expected outputs in the problem table.

### rtl/data_serializer.sv
```verilog
module data_serializer #(
  parameter DATA_W   = 8,
  parameter BIT_ORDER= 0,
  parameter PARITY   = 0
)(
  input                  clk,
  input                  reset,
  // Parallel data interface
  input                  p_valid_i,
  input  [DATA_W-1:0]    p_data_i,
  output                 p_ready_o,
  // Serial data interface
  output                 s_valid_o,
  output                 s_data_o,
  input                  s_ready_i,
  // Transmission enable
  input                  tx_en_i
);

  // Local constants
  localparam ST_RX = 1'b0;
  localparam ST_TX = 1'b1;

  // If PARITY != 0 => We use 1 extra bit for parity
  localparam EXTRA_BIT = (PARITY == 0) ? 0 : 1;
  localparam SHIFT_W   = DATA_W + EXTRA_BIT;

  // Internal regs
  reg                    state_q, state_d;
  reg  [SHIFT_W-1:0]     shift_reg_q, shift_reg_d;
  reg  [$clog2(SHIFT_W)-1:0] count_q, count_d;

  // Parity calculation
  wire parity_bit_even = ^p_data_i;   // XOR => "even"
  wire parity_bit_odd  = ~^p_data_i;  // invert XOR => "odd"

  wire parity_bit = (PARITY == 1) ? parity_bit_even :
                    (PARITY == 2) ? parity_bit_odd  :
                                    1'b0; // NONE

  // Sequential state & register updates
  always @(posedge clk or posedge reset) begin
    if (reset) begin
      state_q     <= ST_RX;
      shift_reg_q <= {SHIFT_W{1'b0}};
      count_q     <= 0;
    end else begin
      state_q     <= state_d;
      shift_reg_q <= shift_reg_d;
      count_q     <= count_d;
    end
  end

  // Next-state logic
  always @* begin
    // Default assignments
    state_d     = state_q;
    shift_reg_d = shift_reg_q;
    count_d     = count_q;

    case (state_q)

      // ST_RX: Load parallel data + parity
      ST_RX: begin
        if (p_valid_i) begin
          if (BIT_ORDER == 0) begin
            // LSB-first => store LSB in shift_reg_d[0]
            if (EXTRA_BIT == 1)
              shift_reg_d = {parity_bit, p_data_i};  // 9 bits if PARITY!=0
            else
              shift_reg_d = p_data_i;                // 8 bits if PARITY=0
          end
          else begin
            // MSB-first => store MSB in shift_reg_d[SHIFT_W-1]
            // Parity appended as LSB when EXTRA_BIT=1
            if (EXTRA_BIT == 1)
              shift_reg_d = {p_data_i, parity_bit};
            else
              shift_reg_d = p_data_i;
          end

          count_d = 0;
          state_d = ST_TX;
        end
      end

      // ST_TX: Shift bits out until SHIFT_W done
      ST_TX: begin
        // Only shift if s_ready_i & tx_en_i
        if (s_ready_i && tx_en_i) begin
          if (count_q == (SHIFT_W - 1)) begin
            // Done sending SHIFT_W bits
            state_d   = ST_RX;
            count_d   = 0;
          end
          else begin
            if (BIT_ORDER == 1) begin
              // MSB-first => shift left by 1
              shift_reg_d = {shift_reg_q[SHIFT_W-2:0], 1'b0};
              count_d     = count_q + 1;
            end
            else begin
              // LSB-first => shift right by 1
              shift_reg_d = {1'b0, shift_reg_q[SHIFT_W-1:1]};
              count_d     = count_q + 1;
            end
          end
        end
      end

      default: begin
        state_d = ST_RX; // safe fallback
      end
    endcase
  end

  // Outputs
  assign s_valid_o = (state_q == ST_TX);

  // LSB-first => s_data_o = shift_reg_q[0]
  // MSB-first => s_data_o = shift_reg_q[SHIFT_W-1]
  assign s_data_o  = (BIT_ORDER == 0) ? shift_reg_q[0] : shift_reg_q[SHIFT_W-1];

  assign p_ready_o = (state_q == ST_RX);

endmodule
```
