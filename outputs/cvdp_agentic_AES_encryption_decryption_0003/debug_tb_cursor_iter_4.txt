## Rationale
- The harness still reports a pure logic mismatch at 140 ns (`DUT o_data = 0xe21e3572…` vs model `0x73f5fe38…`); the build succeeds, so the remaining bug is in multi-cycle AES/key-schedule behavior, not elaboration.
- Cocotb/VPI cannot probe named internal blocks (`next_data`, `rk_source_sel`, `reset_regs`, etc.), so the harness gives only the final ciphertext with no visibility into `round_ff`, `rk_source`, or `expanded_key_ff` during the 11-round pipeline.
- The stimulus is simple and deterministic to mirror (10 ns clock, async reset, one-cycle `i_start`/`i_update_key`, poll `o_done`); an iverilog TB can reproduce that flow and dump waves on DUT hierarchy signals the harness cannot see.
- Prior fix iterations targeted key-expansion chaining and `rk_source` timing; waves are needed to confirm whether RK0–RK10 are correct on the start cycle and frozen/stable through rounds 1–10.
- A focused local sim is cheap here (single RTL file, no UVM/DPI) and directly supports the next minimal RTL fix, unlike another blind output-only iteration.

module tb_debug;
    reg         clk;
    reg         rst_async_n;
    reg         i_update_key;
    reg  [127:0] i_key;
    reg         i_start;
    reg  [127:0] i_data;
    wire        o_done;
    wire [127:0] o_data;

    // NIST AES-128 ECB vector (always valid reference)
    localparam [127:0] NIST_KEY  = 128'h000102030405060708090a0b0c0d0e0f;
    localparam [127:0] NIST_DATA = 128'h00112233445566778899aabbccddeeff;
    localparam [127:0] NIST_EXP  = 128'h69c4e0d86a7b0430d8cdb78070b4c55a;

    // Cocotb-style first run after reset: j=0, update_key=1
    // Replace with seed-specific values if harness seed is known
    localparam [127:0] CVDP_KEY  = 128'h202cb14d3b459f03f38ef4eafc644826;
    localparam [127:0] CVDP_DATA = 128'h32a89dcebf9b590c19c1118bbd4fd8f7;
    localparam [127:0] CVDP_EXP  = 128'h5e02dca5cab206192f61a8467073de64;

    reg  [127:0] exp_data;
    reg          check_en;
    reg          saw_busy;
    wire         tb_mismatch;
    integer      scenario;

    aes128_encrypt dut (
        .clk          (clk),
        .rst_async_n  (rst_async_n),
        .i_update_key (i_update_key),
        .i_key        (i_key),
        .i_start      (i_start),
        .i_data       (i_data),
        .o_done       (o_done),
        .o_data       (o_data)
    );

    initial clk = 1'b0;
    always #5 clk = ~clk;

    initial begin
        $dumpfile("wave.vcd");
        $dumpvars(0, tb_debug);
    end

    assign tb_mismatch = check_en && o_done && saw_busy && (o_data !== exp_data);

    task pulse_reset;
        begin
            i_start      = 1'b0;
            i_update_key = 1'b0;
            i_key        = 128'h0;
            i_data       = 128'h0;
            rst_async_n  = 1'b0;
            @(posedge clk);
            rst_async_n  = 1'b1;
            @(posedge clk);
        end
    endtask

    // Matches test_aes128_encrypt.py sequence
    task run_encrypt;
        input        update_key;
        input [127:0] key;
        input [127:0] data;
        input [127:0] expected;
        begin
            check_en     = 1'b0;
            saw_busy     = 1'b0;
            exp_data     = expected;

            i_update_key = update_key;
            i_key        = key;
            i_data       = data;
            i_start      = 1'b1;

            $display("[%0t] scenario=%0d start key=%h data=%h update_key=%b",
                     $time, scenario, key, data, update_key);

            @(posedge clk);

            i_start      = 1'b0;
            i_update_key = 1'b0;
            i_key        = 128'h0;
            i_data       = 128'h0;

            @(posedge clk);
            check_en = 1'b1;

            while (!o_done) begin
                saw_busy = 1'b1;
                $display("[%0t] busy round_ff=%0d rk_msb=%h ff_msb=%h nx_msb=%h",
                         $time,
                         dut.round_ff,
                         dut.rk_source[127:0],
                         dut.expanded_key_ff[127:0],
                         dut.expanded_key_nx[127:0]);
                @(posedge clk);
            end

            if (o_data !== expected) begin
                $display("First mismatch occurred at time %0t", $time);
                $display("  scenario=%0d o_data=%h expected=%h", scenario, o_data, expected);
            end else begin
                $display("[%0t] scenario=%0d PASS o_data=%h", $time, scenario, o_data);
            end

            check_en = 1'b0;
            saw_busy = 1'b0;
            scenario = scenario + 1;
            repeat (2) @(posedge clk);
        end
    endtask

    always @(posedge clk) begin
        if (tb_mismatch)
            $display("tb_mismatch flagged at time %0t (scenario=%0d)", $time, scenario);
    end

    initial begin
        scenario = 0;

        pulse_reset();

        // Deterministic sanity check
        run_encrypt(1'b1, NIST_KEY, NIST_DATA, NIST_EXP);

        // Cocotb-equivalent first post-reset transaction
        pulse_reset();
        run_encrypt(1'b1, CVDP_KEY, CVDP_DATA, CVDP_EXP);

        #50;
        $finish;
    end
endmodule
