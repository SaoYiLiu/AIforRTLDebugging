## Rationale
- **Line 12:** `reg [7:0] buf [0:8];` is a second unpacked array in the same module as `pixel_buffer[0:8]`. Icarus Verilog 13 rejects this with `syntax error` / `Syntax error in variable list` (confirmed locally with `iverilog -g2012`, exit status 24), matching the harness compile failure.
- **Lines 34–42:** Blocking assignments to `buf[...]` fail as cascading errors because `buf` never elaborates; Icarus reports `Syntax in assignment statement l-value` on those lines.
- **Lines 44–56:** The `$signed(...)` gradient expressions then fail with `Malformed statement` for the same reason — they depend on the invalid `buf` array.
- **Fix:** Remove the `buf` array entirely. When `pixel_count == 4'd8`, the 9th pixel is `pixel_in` and the prior eight pixels are already in `pixel_buffer[7:0]` (`pixel_buffer[7]` = top-left through `pixel_buffer[0]` = bottom-middle).
- **Lines 44–57 (revised):** Compute Gx/Gy directly from `pixel_buffer[7:0]` and `pixel_in`, preserving correct Sobel kernel mapping without a second unpacked array.
- **Verified:** The revised design compiles under `iverilog -g2012` and passes all six specification test cases (valid on cycle 9, cleared on cycle 10, correct `edge_out` values).

```verilog
module sobel_filter (
    input            clk,
    input            rst_n,
    input      [7:0] pixel_in,
    input            valid_in,
    output reg [7:0] edge_out,
    output reg       valid_out
);
    reg signed [10:0] Gx, Gy;
    reg signed [10:0] gx_val, gy_val, mag;
    reg [7:0] pixel_buffer[0:8];
    reg [3:0] pixel_count;
    integer i;

    parameter THRESHOLD = 11'd128;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            for (i = 0; i < 9; i = i + 1) begin
                pixel_buffer[i] <= 8'd0;
            end
            pixel_count <= 4'd0;
            Gx        <= 11'sd0;
            Gy        <= 11'sd0;
            edge_out  <= 8'd0;
            valid_out <= 1'b0;
        end else begin
            valid_out <= 1'b0;
            edge_out  <= 8'd0;

            if (valid_in) begin
                if (pixel_count == 4'd8) begin
                    gx_val = -$signed({3'b0, pixel_buffer[7]})
                           - ($signed({3'b0, pixel_buffer[4]}) << 1)
                           -  $signed({3'b0, pixel_buffer[1]})
                           +  $signed({3'b0, pixel_buffer[5]})
                           + ($signed({3'b0, pixel_buffer[2]}) << 1)
                           +  $signed({3'b0, pixel_in});

                    gy_val = -$signed({3'b0, pixel_buffer[7]})
                           - ($signed({3'b0, pixel_buffer[6]}) << 1)
                           -  $signed({3'b0, pixel_buffer[5]})
                           +  $signed({3'b0, pixel_buffer[1]})
                           + ($signed({3'b0, pixel_buffer[0]}) << 1)
                           +  $signed({3'b0, pixel_in});

                    mag = (gx_val < 0 ? -gx_val : gx_val)
                        + (gy_val < 0 ? -gy_val : gy_val);

                    Gx        <= gx_val;
                    Gy        <= gy_val;
                    edge_out  <= (mag > THRESHOLD) ? 8'd255 : 8'd0;
                    valid_out <= 1'b1;

                    for (i = 0; i < 9; i = i + 1) begin
                        pixel_buffer[i] <= 8'd0;
                    end
                    pixel_count <= 4'd0;
                end else begin
                    pixel_buffer[8] <= pixel_buffer[7];
                    pixel_buffer[7] <= pixel_buffer[6];
                    pixel_buffer[6] <= pixel_buffer[5];
                    pixel_buffer[5] <= pixel_buffer[4];
                    pixel_buffer[4] <= pixel_buffer[3];
                    pixel_buffer[3] <= pixel_buffer[2];
                    pixel_buffer[2] <= pixel_buffer[1];
                    pixel_buffer[1] <= pixel_buffer[0];
                    pixel_buffer[0] <= pixel_in;
                    pixel_count <= pixel_count + 1'b1;
                end
            end else begin
                for (i = 0; i < 9; i = i + 1) begin
                    pixel_buffer[i] <= 8'd0;
                end
                pixel_count <= 4'd0;
            end
        end
    end
endmodule
```
