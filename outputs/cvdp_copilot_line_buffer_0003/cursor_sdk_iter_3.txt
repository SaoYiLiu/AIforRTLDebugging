## Rationale
- **Row/column indexing is already correct** in the current candidate (lines 39–96 use `i_image_row_start + row` and `i_image_col_start + col`). The remaining failures are in border handling, not window traversal.
- **Mirror column indexing is broken for OOB pixels** (line 81): `NS_COLUMN-1-(i_image_col_start + col)` goes negative when `i_image_col_start + col >= NS_COLUMN` (e.g. index 3 with `NS_COLUMN=3` yields −1). In unsigned Verilog that wraps to a huge index, producing corrupted pixels. The spec mirror example (`0x484d4d`) requires index 3 → column 1 (`0x48`), which needs the distance-based reflect formula instead.
- **Mirror row indexing has the same class of bug** (line 79): `2*NS_ROW-1-(i_image_row_start + row)` maps row index 3 to row 2 when it should map to row 1 (reflect sequence `0,1,2,1,0,…`). Harness mismatches like `0x29727eb6… != 0xb6f07e7d…` show wrong values in upper window rows while lower bytes match, consistent with incorrect row mirroring.
- **Extend/wrap/mirror use `if` / `else if` chains** (lines 65–97), so when both row and column are OOB only the row branch runs. Extend mode then reads `image_buffer_ff[NS_ROW-1][i_image_col_start + col]` with an out-of-range column (line 66), instead of clamping both dimensions to `[NS_ROW-1][NS_COLUMN-1]`.
- **Harness evidence:** Failures at 310 ns with parameters like `NS_ROW=10, NS_C_OUT=3` show partial matches (e.g. suffix `…7cf52c0ffa56` identical, prefix wrong), which fits a bug that only affects certain OOB window positions rather than all pixels or packing.
- **Input unpacking is correct** (lines 115–117): the `row_image[NS_COLUMN-col-1]` reversal combined with LSB-first slicing places `0xa6484d` into buffer columns `[0xa6, 0x48, 0x4d]` per the spec row-storing example — no change needed there.
- **Fix:** Refactor `window_assignment` to compute `src_row`/`src_col` per pixel, apply row and column boundary transforms independently for modes 2–4, and use the distance-based reflect formula for mirror mode on both axes.

### rtl/line_buffer.sv
```verilog
module line_buffer #(
    parameter NBW_DATA  = 'd8,  // Bit width of grayscale input/output data
    parameter NS_ROW    = 'd10, // Number of rows
    parameter NS_COLUMN = 'd8,  // Number of columns
    parameter NBW_ROW   = 'd4,  // log2(NS_ROW). Bit width of i_image_row_start
    parameter NBW_COL   = 'd3,  // log2(NS_COLUMN). Bit width of i_image_col_start
    parameter NBW_MODE  = 'd3,  // Bit width of mode input
    parameter NS_R_OUT  = 'd4,  // Number of rows of the output window
    parameter NS_C_OUT  = 'd3,  // Number of columns of the output window
    parameter CONSTANT  = 'd255 // Constant value to use in PAD_CONSTANT mode
) (
    input  logic                                  clk,
    input  logic                                  rst_async_n,
    input  logic [NBW_MODE-1:0]                   i_mode,
    input  logic                                  i_valid,
    input  logic                                  i_update_window,
    input  logic [NBW_DATA*NS_COLUMN-1:0]         i_row_image,
    input  logic [NBW_ROW-1:0]                    i_image_row_start,
    input  logic [NBW_COL-1:0]                    i_image_col_start,
    output logic [NBW_DATA*NS_R_OUT*NS_C_OUT-1:0] o_image_window
);

// ----------------------------------------
// - Wires/Registers creation
// ----------------------------------------
logic [NBW_DATA-1:0] image_buffer_ff [NS_ROW][NS_COLUMN];
logic [NBW_DATA-1:0] row_image [NS_COLUMN];
logic [NBW_DATA-1:0] window [NS_R_OUT][NS_C_OUT];
logic [NBW_DATA*NS_R_OUT*NS_C_OUT-1:0] image_window_ff;

// ----------------------------------------
// - Output generation
// ----------------------------------------
always_comb begin : window_assignment
    for (int row = 0; row < NS_R_OUT; row++) begin
        for (int col = 0; col < NS_C_OUT; col++) begin
            int src_row;
            int src_col;
            int row_dist;
            int col_dist;

            src_row = i_image_row_start + row;
            src_col = i_image_col_start + col;

            case (i_mode)
                3'd0: begin // NO_BOUND_PROCESS
                    if (src_row >= NS_ROW || src_col >= NS_COLUMN)
                        window[row][col] = 0;
                    else
                        window[row][col] = image_buffer_ff[src_row][src_col];
                end
                3'd1: begin // PAD_CONSTANT
                    if (src_row >= NS_ROW || src_col >= NS_COLUMN)
                        window[row][col] = CONSTANT;
                    else
                        window[row][col] = image_buffer_ff[src_row][src_col];
                end
                3'd2: begin // EXTEND_NEAR
                    if (src_row >= NS_ROW)
                        src_row = NS_ROW - 1;
                    if (src_col >= NS_COLUMN)
                        src_col = NS_COLUMN - 1;
                    window[row][col] = image_buffer_ff[src_row][src_col];
                end
                3'd3: begin // MIRROR_BOUND
                    if (src_row >= NS_ROW) begin
                        row_dist = src_row - (NS_ROW - 1);
                        if (row_dist < (NS_ROW - 1))
                            src_row = (NS_ROW - 1) - row_dist;
                        else
                            src_row = 2 * (NS_ROW - 1) - row_dist;
                    end
                    if (src_col >= NS_COLUMN) begin
                        col_dist = src_col - (NS_COLUMN - 1);
                        if (col_dist < (NS_COLUMN - 1))
                            src_col = (NS_COLUMN - 1) - col_dist;
                        else
                            src_col = 2 * (NS_COLUMN - 1) - col_dist;
                    end
                    window[row][col] = image_buffer_ff[src_row][src_col];
                end
                3'd4: begin // WRAP_AROUND
                    if (src_row >= NS_ROW)
                        src_row = src_row - NS_ROW;
                    if (src_col >= NS_COLUMN)
                        src_col = src_col - NS_COLUMN;
                    window[row][col] = image_buffer_ff[src_row][src_col];
                end
                default: begin
                    window[row][col] = 0;
                end
            endcase
        end
    end
end

// ----------------------------------------
// - Input control
// ----------------------------------------
generate
    for (genvar col = 0; col < NS_COLUMN; col++) begin : unpack_row_image
        assign row_image[NS_COLUMN-col-1] = i_row_image[(col+1)*NBW_DATA-1-:NBW_DATA];
    end
endgenerate

always_ff @(posedge clk or negedge rst_async_n) begin : ctrl_regs
    if(~rst_async_n) begin
        image_window_ff <= 0;
        for (int row = 0; row < NS_ROW; row++) begin
            for (int col = 0; col < NS_COLUMN; col++) begin
                image_buffer_ff[row][col] <= 0;
            end
        end
    end else begin
        if(i_valid) begin
            for (int col = 0; col < NS_COLUMN; col++) begin
                image_buffer_ff[0][col] <= row_image[col];
            end

            for (int row = 1; row < NS_ROW; row++) begin
                for (int col = 0; col < NS_COLUMN; col++) begin
                    image_buffer_ff[row][col] <= image_buffer_ff[row-1][col];
                end
            end
        end

        if(i_update_window) begin
            image_window_ff <= o_image_window;
        end
    end
end

// ----------------------------------------
// - Output packing
// ----------------------------------------
generate
    for(genvar row = 0; row < NS_R_OUT; row++) begin : out_row
        for(genvar col = 0; col < NS_C_OUT; col++) begin : out_col
            always_comb begin
                if(i_update_window) begin
                    o_image_window[(row*NS_C_OUT+col+1)*NBW_DATA-1-:NBW_DATA] = window[row][col];
                end else begin
                    o_image_window[(row*NS_C_OUT+col+1)*NBW_DATA-1-:NBW_DATA] = image_window_ff[(row*NS_C_OUT+col+1)*NBW_DATA-1-:NBW_DATA];
                end
            end
        end
    end
endgenerate

endmodule : line_buffer
```
