## Rationale
- **Root cause:** In `window_assignment` (lines 34–109), the loop indices `row` and `col` are swapped when computing buffer addresses and out-of-range checks. The spec requires `window[row][col]` to map to `image_buffer_ff[i_image_row_start + row][i_image_col_start + col]`, but the RTL uses `image_buffer_ff[i_image_row_start + col][i_image_col_start + row]` (lines 44, 57, 70, 83, 96).
- **Boundary checks are also swapped:** Row OOB is tested with `i_image_row_start + col >= NS_ROW` (line 39) instead of `i_image_row_start + row >= NS_ROW`, and column OOB with `i_image_col_start + row >= NS_COLUMN` (line 41) instead of `i_image_col_start + col >= NS_COLUMN`. This affects all five valid modes (0–4).
- **Cycle 5 evidence:** With mode 1 (PAD_CONSTANT), `i_image_row_start=0`, `i_image_col_start=2`, observed `0xffff000e` vs expected `0xff00ff0e`. Simulation of the buggy indexing reproduces `0xffff000e` exactly; correcting the swap yields `0xff00ff0e`.
- **Cycles 8 and 10:** Mirror mode (cycle 8: `0xe0ef9f9` vs `0xef90ef9`) and extend-nearest mode (cycle 10: `0xd5d80baa` vs `0xd50bd8aa`) fail for the same reason — wrong source pixels are selected when indices are transposed.
- **Harness pattern:** Failures like `0xffffffffffff000000000000 != 0xff0000ff0000ff0000ff0000` show CONSTANT padding (`0xff`) landing in wrong window positions, consistent with transposed row/column traversal rather than wrong mode logic or packing.
- **Not the bug:** Row storage (lines 129–138), input unpacking (lines 114–117), output packing (lines 150–162), and `i_update_window` hold behavior are consistent with the spec; only the window-extraction indexing is wrong.
- **Fix:** In every mode branch, replace `col` with `row` and `row` with `col` in OOB conditions and all `image_buffer_ff[...]` index expressions (including mirror `2*NS_ROW-1-(...)` and wrap `(...)-NS_ROW/NS_COLUMN` terms).

module line_buffer #(
    parameter NBW_DATA  = 'd8,  // Bit width of grayscale input/output data
    parameter NS_ROW    = 'd10, // Number of rows
    parameter NS_COLUMN = 'd8,  // Number of columns
    parameter NBW_ROW   = 'd4,  // log2(NS_ROW). Bit width of i_image_row_start
    parameter NBW_COL   = 'd3,  // log2(NS_COLUMN). Bit width of i_image_col_start
    parameter NBW_MODE  = 'd3,  // Bit width of mode input
    parameter NS_R_OUT  = 'd4,  // Number of rows of the output window
    parameter NS_C_OUT  = 'd3,  // Number of columns of the output window
    parameter CONSTANT  = 'd255 // Constant value to use in PAD_CONSTANT mode
) (
    input  logic                                  clk,
    input  logic                                  rst_async_n,
    input  logic [NBW_MODE-1:0]                   i_mode,
    input  logic                                  i_valid,
    input  logic                                  i_update_window,
    input  logic [NBW_DATA*NS_COLUMN-1:0]         i_row_image,
    input  logic [NBW_ROW-1:0]                    i_image_row_start,
    input  logic [NBW_COL-1:0]                    i_image_col_start,
    output logic [NBW_DATA*NS_R_OUT*NS_C_OUT-1:0] o_image_window
);

// ----------------------------------------
// - Wires/Registers creation
// ----------------------------------------
logic [NBW_DATA-1:0] image_buffer_ff [NS_ROW][NS_COLUMN];
logic [NBW_DATA-1:0] row_image [NS_COLUMN];
logic [NBW_DATA-1:0] window [NS_R_OUT][NS_C_OUT];
logic [NBW_DATA*NS_R_OUT*NS_C_OUT-1:0] image_window_ff;

// ----------------------------------------
// - Output generation
// ----------------------------------------
always_comb begin : window_assignment
    case(i_mode)
        3'd0: begin // NO_BOUND_PROCESS
            for (int row = 0; row < NS_R_OUT; row++) begin
                for (int col = 0; col < NS_C_OUT; col++) begin
                    if(i_image_row_start + row >= NS_ROW) begin
                        window[row][col] = 0;
                    end else if (i_image_col_start + col >= NS_COLUMN) begin
                        window[row][col] = 0;
                    end else begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][i_image_col_start + col];
                    end
                end
            end
        end
        3'd1: begin // PAD_CONSTANT
            for (int row = 0; row < NS_R_OUT; row++) begin
                for (int col = 0; col < NS_C_OUT; col++) begin
                    if(i_image_row_start + row >= NS_ROW) begin
                        window[row][col] = CONSTANT;
                    end else if (i_image_col_start + col >= NS_COLUMN) begin
                        window[row][col] = CONSTANT;
                    end else begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][i_image_col_start + col];
                    end
                end
            end
        end
        3'd2: begin // EXTEND_NEAR
            for (int row = 0; row < NS_R_OUT; row++) begin
                for (int col = 0; col < NS_C_OUT; col++) begin
                    if(i_image_row_start + row >= NS_ROW) begin
                        window[row][col] = image_buffer_ff[NS_ROW-1][i_image_col_start + col];
                    end else if (i_image_col_start + col >= NS_COLUMN) begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][NS_COLUMN-1];
                    end else begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][i_image_col_start + col];
                    end
                end
            end
        end
        3'd3: begin // MIRROR_BOUND
            for (int row = 0; row < NS_R_OUT; row++) begin
                for (int col = 0; col < NS_C_OUT; col++) begin
                    if(i_image_row_start + row >= NS_ROW) begin
                        window[row][col] = image_buffer_ff[2*NS_ROW-1-(i_image_row_start + row)][i_image_col_start + col];
                    end else if (i_image_col_start + col >= NS_COLUMN) begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][NS_COLUMN-1-(i_image_col_start + col)];
                    end else begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][i_image_col_start + col];
                    end
                end
            end
        end
        3'd4: begin // WRAP_AROUND
            for (int row = 0; row < NS_R_OUT; row++) begin
                for (int col = 0; col < NS_C_OUT; col++) begin
                    if(i_image_row_start + row >= NS_ROW) begin
                        window[row][col] = image_buffer_ff[(i_image_row_start + row)-NS_ROW][i_image_col_start + col];
                    end else if (i_image_col_start + col >= NS_COLUMN) begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][(i_image_col_start + col)-NS_COLUMN];
                    end else begin
                        window[row][col] = image_buffer_ff[i_image_row_start + row][i_image_col_start + col];
                    end
                end
            end
        end
        default: begin
            for (int row = 0; row < NS_R_OUT; row++) begin
                for (int col = 0; col < NS_C_OUT; col++) begin
                    window[row][col] = 0;
                end
            end
        end
    endcase
end

// ----------------------------------------
// - Input control
// ----------------------------------------
generate
    for (genvar col = 0; col < NS_COLUMN; col++) begin : unpack_row_image
        assign row_image[NS_COLUMN-col-1] = i_row_image[(col+1)*NBW_DATA-1-:NBW_DATA];
    end
endgenerate

always_ff @(posedge clk or negedge rst_async_n) begin : ctrl_regs
    if(~rst_async_n) begin
        image_window_ff <= 0;
        for (int row = 0; row < NS_ROW; row++) begin
            for (int col = 0; col < NS_COLUMN; col++) begin
                image_buffer_ff[row][col] <= 0;
            end
        end
    end else begin
        if(i_valid) begin
            for (int col = 0; col < NS_COLUMN; col++) begin
                image_buffer_ff[0][col] <= row_image[col];
            end

            for (int row = 1; row < NS_ROW; row++) begin
                for (int col = 0; col < NS_COLUMN; col++) begin
                    image_buffer_ff[row][col] <= image_buffer_ff[row-1][col];
                end
            end
        end

        if(i_update_window) begin
            image_window_ff <= o_image_window;
        end
    end
end

// ----------------------------------------
// - Output packing
// ----------------------------------------
generate
    for(genvar row = 0; row < NS_R_OUT; row++) begin : out_row
        for(genvar col = 0; col < NS_C_OUT; col++) begin : out_col
            always_comb begin
                if(i_update_window) begin
                    o_image_window[(row*NS_C_OUT+col+1)*NBW_DATA-1-:NBW_DATA] = window[row][col];
                end else begin
                    o_image_window[(row*NS_C_OUT+col+1)*NBW_DATA-1-:NBW_DATA] = image_window_ff[(row*NS_C_OUT+col+1)*NBW_DATA-1-:NBW_DATA];
                end
            end
        end
    end
endgenerate

endmodule : line_buffer
