## Rationale
- The harness only reports a final `o_data` mismatch at 140 ns (`0x2e27a813…` vs `0xe3ae342f…`); it does not show whether key expansion, round-key selection, or the cipher datapath is wrong after the recent `aes_sbox` / `key_expand_src` changes.
- The failure is multi-cycle (11 AES rounds plus key-update handshake), so a VCD trace of `round_ff`, `rk_source`, `round_key_sched`, `key_expand_src`, and `expanded_key_computed` is the fastest way to localize the bug to schedule vs cipher logic.
- Cocotb’s first failing case is always `j=0` with `i_update_key=1` right after reset — that exact protocol can be mirrored in iverilog without randomness, using the NIST vector as a golden reference and a fixed secondary vector for regression.
- Prior debug already showed zero RK0 and corrupted RK1 when expansion was broken; a local sim is needed to confirm whether those internal signals are now correct or a new bug remains (e.g., `round_key_sched` snapshot timing, stored-key reuse with `i_update_key=0`).
- The design is a single module with a simple port list — a faithful mini-TB is straightforward and iverilog-friendly, unlike wide-bus or UVM environments where local sim adds little value.

## DebugTestbench

```verilog
`timescale 1ns/1ns

module tb_debug;
    logic clk;
    logic rst_async_n;
    logic i_update_key;
    logic [127:0] i_key;
    logic i_start;
    logic [127:0] i_data;
    logic o_done;
    logic [127:0] o_data;

    // Scenario control
    integer scenario;
    logic [127:0] exp_data;
    logic [127:0] exp_rk0, exp_rk1, exp_rk10;
    logic chk_enable;
    logic tb_mismatch;

    aes128_encrypt dut (
        .clk           (clk),
        .rst_async_n   (rst_async_n),
        .i_update_key  (i_update_key),
        .i_key         (i_key),
        .i_start       (i_start),
        .i_data        (i_data),
        .o_done        (o_done),
        .o_data        (o_data)
    );

    // 10 ns period, matching cocotb harness
    initial clk = 1'b0;
    always #5 clk = ~clk;

    initial begin
        $dumpfile("wave.vcd");
        $dumpvars(0, tb_debug);
    end

    // Flag first output mismatch while checking is enabled
    assign tb_mismatch = chk_enable && o_done && (o_data !== exp_data);

    always @(posedge clk) begin
        if (tb_mismatch) begin
            $display("First mismatch occurred at time %0t", $time);
            $display("[%0t] scenario=%0d round_ff=%0d o_done=%0d o_data=%h expected=%h",
                     $time, scenario, dut.round_ff, o_done, o_data, exp_data);
            $display("  key_expand_src=%h valid_key=%h i_update_key=%b i_start=%b",
                     dut.key_expand_src, dut.valid_key, i_update_key, i_start);
            $display("  rk0: ff=%h comp=%h rk_src=%h sched=%h",
                     dut.expanded_key_ff[1407:1280],
                     dut.expanded_key_computed[1407:1280],
                     dut.rk_source[1407:1280],
                     dut.round_key_sched[1407:1280]);
            $display("  rk1: ff=%h comp=%h rk_src=%h sched=%h",
                     dut.expanded_key_ff[1279:1152],
                     dut.expanded_key_computed[1279:1152],
                     dut.rk_source[1279:1152],
                     dut.round_key_sched[1279:1152]);
            $display("  rk10: ff=%h comp=%h rk_src=%h sched=%h",
                     dut.expanded_key_ff[127:0],
                     dut.expanded_key_computed[127:0],
                     dut.rk_source[127:0],
                     dut.round_key_sched[127:0]);
        end
    end

    task automatic cocotb_reset;
        input logic [127:0] key;
        input logic [127:0] data;
        input logic         upd;
        integer timeout;
        begin
            i_start      = 1'b0;
            i_update_key = 1'b0;
            i_key        = '0;
            i_data       = '0;
            rst_async_n  = 1'b0;
            @(posedge clk);
            rst_async_n  = 1'b1;
            @(posedge clk);

            i_update_key = upd;
            i_key        = key;
            i_data       = data;
            i_start      = 1'b1;
            @(posedge clk);

            i_start      = 1'b0;
            i_key        = '0;
            i_update_key = 1'b0;
            i_data       = '0;
            @(posedge clk);

            timeout = 0;
            while (!o_done) begin
                @(posedge clk);
                timeout = timeout + 1;
                if (timeout > 20) begin
                    $display("TIMEOUT waiting for o_done at t=%0t scenario=%0d", $time, scenario);
                    $finish;
                end
            end
        end
    endtask

    initial begin
        chk_enable   = 1'b0;
        scenario     = 0;
        rst_async_n  = 1'b0;
        i_start      = 1'b0;
        i_update_key = 1'b0;
        i_key        = '0;
        i_data       = '0;

        // Allow reset to settle
        repeat (2) @(posedge clk);
        rst_async_n = 1'b1;
        @(posedge clk);

        // ---------------------------------------------------------------
        // Scenario 0: NIST AES-128 vector (cocotb j=0 style: update_key=1)
        // Matches first post-reset encryption in harness (always update_key=1)
        // ---------------------------------------------------------------
        scenario  = 0;
        exp_data  = 128'h69c4e0d86a7b0430d8cdb78070b4c55a;
        exp_rk0   = 128'h000102030405060708090a0b0c0d0e0f;
        exp_rk1   = 128'hd6aa74fdd2af72fadaa678f1d6ab76fe;
        exp_rk10  = 128'h13111d7fe3944a17f307a78b4d2b30c5;
        chk_enable = 1'b1;

        cocotb_reset(
            128'h000102030405060708090a0b0c0d0e0f,
            128'h00112233445566778899aabbccddeeff,
            1'b1
        );

        if (o_data !== exp_data) begin
            $display("FAIL scenario %0d at t=%0t", scenario, $time);
            $finish(1);
        end else begin
            $display("PASS scenario %0d NIST vector at t=%0t o_data=%h", scenario, $time, o_data);
        end

        // Sanity-check expanded schedule MSBs while still at o_done
        if (dut.expanded_key_computed[1407:1280] !== exp_rk0) begin
            $display("WARN scenario %0d RK0 comp=%h expected=%h", scenario,
                     dut.expanded_key_computed[1407:1280], exp_rk0);
        end
        if (dut.expanded_key_computed[1279:1152] !== exp_rk1) begin
            $display("WARN scenario %0d RK1 comp=%h expected=%h", scenario,
                     dut.expanded_key_computed[1279:1152], exp_rk1);
        end
        if (dut.expanded_key_computed[127:0] !== exp_rk10) begin
            $display("WARN scenario %0d RK10 comp=%h expected=%h", scenario,
                     dut.expanded_key_computed[127:0], exp_rk10);
        end

        // ---------------------------------------------------------------
        // Scenario 1: Encrypt again WITHOUT key update (stored schedule)
        // Cocotb j>0 often uses update_key=0; catches round_key_sched bugs
        // ---------------------------------------------------------------
        scenario  = 1;
        exp_data  = 128'h69c4e0d86a7b0430d8cdb78070b4c55a; // same key => same PT => same CT

        cocotb_reset(
            128'hffffffffffffffffffffffffffffffff, // ignored when update_key=0
            128'h00112233445566778899aabbccddeeff,
            1'b0
        );

        if (o_data !== exp_data) begin
            $display("FAIL scenario %0d stored-key reuse at t=%0t o_data=%h", scenario, $time, o_data);
            $finish(1);
        end else begin
            $display("PASS scenario %0d stored-key reuse at t=%0t", scenario, $time);
        end

        // ---------------------------------------------------------------
        // Scenario 2: New key, update_key=1 (post-reset style second block)
        // Fixed vector for waveform compare if NIST passes but harness random fails
        // ---------------------------------------------------------------
        scenario  = 2;
        exp_data  = 128'h3ad77bb40d7a3660a89ecaf32466ef97; // precomputed AES-128 reference

        cocotb_reset(
            128'h2b7e151628aed2a6abf7158809cf4f3c,
            128'h3243f6a8885a308d313198a2e0370734,
            1'b1
        );

        if (o_data !== exp_data) begin
            $display("FAIL scenario %0d second vector at t=%0t o_data=%h expected=%h",
                     scenario, $time, o_data, exp_data);
            $finish(1);
        end else begin
            $display("PASS scenario %0d second vector at t=%0t o_data=%h", scenario, $time, o_data);
        end

        chk_enable = 1'b0;
        $display("ALL tb_debug scenarios passed at t=%0t", $time);
        $finish;
    end
endmodule
```

**Usage:** compile with `iverilog -g2012 -o sim.vvp rtl/aes128_encrypt.sv tb_debug.sv && vvp sim.vvp`, then inspect `wave.vcd` focusing on `round_ff`, `key_expand_src`, `expanded_key_computed[1407:1280]` (RK0), `expanded_key_computed[1279:1152]` (RK1), and `round_key_sched` at the cycle where `i_start & o_done` is asserted. If scenario 0 fails with RK1 still `62636363…`, key expansion remains broken; if RK values are correct but `o_data` is wrong, debug the cipher round-key indexing path instead.
