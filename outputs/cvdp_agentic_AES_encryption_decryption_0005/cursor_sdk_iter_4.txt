## Rationale
- **InvShiftRows is wrong on rows 1 and 3 (lines 149–152, 159–162)**: AES-128 inverse row 1 must shift **right** by 1 (`s'[1,c]=s[1,c-1]`), but the RTL shifts **left** (`s'[1,0]=s[1,1]`). Row 3 must shift right by 3 (`s'[3,c]=s[3,c+1]`), but the RTL implements a right-by-1 pattern (`s'[3,0]=s[3,3]`). Row 2 is correct. This corrupts InvSubBytes input on every round except the AddRoundKey-only initial step, matching the harness: structured wrong output (`0xd27b4e3c…` vs `0xe90267c3…`) at 240 ns after prior FSM/key-expansion fixes.

- **Iverilog key-expansion `gen_idx` selection is unreliable (lines 61–75 in `aes128_key_expansion.sv`)**: Local debug stderr shows `constant selects in always_* processes are not currently supported` on the `unique case (1'b1)` one-hot decoder—the same simulator the harness uses. If `gen_idx` sticks at 0, only K0/K1 logic runs and round keys K2–K10 are wrong, producing incorrect AddRoundKey on all inner rounds.

- **Prior fixes remain necessary and are retained**: Column-major `(4*j+i)` packing (lines 82, 113, 167), corrected InvMixColumns row wiring (lines 138–141), serial one-key-per-cycle expansion (lines 115–116), and `update_key_ff`/`key_done` wait at `round_ff==1` (lines 60–63, 83–88) are unchanged; the harness still fails at Run 0 with `i_update_key=1`, so the remaining bugs are in InvShiftRows and/or key-index selection, not the FSM skeleton.

- **240 ns completion time confirms multi-cycle flow is alive**: Failure at 240 ns (≈24 × 10 ns clocks) is consistent with 10 key-expansion cycles plus 11 decrypt rounds; the bug is functional in the datapath, not a hang or early termination.

- **Round 1 and round 11 special-casing is still correct (lines 166–171)**: Initial AddRoundKey-only (K10) and final InvShiftRows→InvSubBytes→AddRoundKey without InvMixColumns (K0) match the task spec; only rows 1 and 3 of InvShiftRows need correction for rounds 2–11.

module aes128_key_expansion #(
    parameter NBW_KEY = 'd128,
    parameter NBW_OUT = 'd1408
) (
    input  logic               clk,
    input  logic               rst_async_n,
    input  logic               i_start,
    input  logic [NBW_KEY-1:0] i_key,
    output logic               o_done,
    output logic [NBW_OUT-1:0] o_expanded_key
);

localparam NBW_BYTE = 'd8;
localparam NBW_WORD = 'd32;
localparam STEPS    = 'd10;

logic [NBW_BYTE-1:0] Rcon   [STEPS];
logic [NBW_OUT-1:0]  expanded_key_nx;
logic [NBW_OUT-1:0]  expanded_key_ff;
logic [NBW_KEY-1:0]  next_rkey;
logic [3:0]          gen_idx;
logic [STEPS:0]      key_exp_steps_ff;

logic [NBW_WORD-1:0] RotWord;
logic [NBW_WORD-1:0] SubWord;
logic [NBW_WORD-1:0] RconXor;

assign o_expanded_key = expanded_key_ff;
assign o_done         = key_exp_steps_ff[STEPS];

always_ff @(posedge clk or negedge rst_async_n) begin : reset_regs
    if(~rst_async_n) begin
        expanded_key_ff  <= {NBW_OUT{1'b0}};
        key_exp_steps_ff <= '0;
    end else begin
        expanded_key_ff <= expanded_key_nx;

        if(i_start) begin
            key_exp_steps_ff <= {{(STEPS-1){1'b0}}, 1'b1};
        end else begin
            if(key_exp_steps_ff[STEPS]) begin
                key_exp_steps_ff <= '0;
            end else if(|key_exp_steps_ff[STEPS-1:0]) begin
                key_exp_steps_ff <= key_exp_steps_ff << 1'b1;
            end
        end
    end
end

assign Rcon[0] = 8'h01;
assign Rcon[1] = 8'h02;
assign Rcon[2] = 8'h04;
assign Rcon[3] = 8'h08;
assign Rcon[4] = 8'h10;
assign Rcon[5] = 8'h20;
assign Rcon[6] = 8'h40;
assign Rcon[7] = 8'h80;
assign Rcon[8] = 8'h1b;
assign Rcon[9] = 8'h36;

always_comb begin : gen_idx_select
    gen_idx = 4'd0;
    for(int k = 0; k < STEPS; k++) begin
        if(key_exp_steps_ff[k]) begin
            gen_idx = 4'(k);
        end
    end
end

sbox uu_sbox0 (
    .i_data(RotWord[NBW_WORD-1-:NBW_BYTE]),
    .o_data(SubWord[NBW_WORD-1-:NBW_BYTE])
);

sbox uu_sbox1 (
    .i_data(RotWord[NBW_WORD-NBW_BYTE-1-:NBW_BYTE]),
    .o_data(SubWord[NBW_WORD-NBW_BYTE-1-:NBW_BYTE])
);

sbox uu_sbox2 (
    .i_data(RotWord[NBW_WORD-2*NBW_BYTE-1-:NBW_BYTE]),
    .o_data(SubWord[NBW_WORD-2*NBW_BYTE-1-:NBW_BYTE])
);

sbox uu_sbox3 (
    .i_data(RotWord[NBW_WORD-3*NBW_BYTE-1-:NBW_BYTE]),
    .o_data(SubWord[NBW_WORD-3*NBW_BYTE-1-:NBW_BYTE])
);

always_comb begin : expand_next_rkey
    RotWord = {
        expanded_key_ff[NBW_OUT-(gen_idx+1)*NBW_KEY+NBW_WORD-NBW_BYTE-1-:(NBW_WORD-NBW_BYTE)],
        expanded_key_ff[NBW_OUT-(gen_idx+1)*NBW_KEY+NBW_WORD-1-:NBW_BYTE]
    };
    RconXor = {SubWord[NBW_WORD-1-:NBW_BYTE] ^ Rcon[gen_idx], SubWord[NBW_WORD-NBW_BYTE-1-:(NBW_WORD-NBW_BYTE)]};

    next_rkey[NBW_KEY-1-:NBW_WORD]            = expanded_key_ff[NBW_OUT-(4*gen_idx+3)*NBW_WORD-1-:NBW_WORD] ^ RconXor;
    next_rkey[NBW_KEY-NBW_WORD-1-:NBW_WORD]   = expanded_key_ff[NBW_OUT-(4*gen_idx+2)*NBW_WORD-1-:NBW_WORD] ^ next_rkey[NBW_KEY-1-:NBW_WORD];
    next_rkey[NBW_KEY-2*NBW_WORD-1-:NBW_WORD] = expanded_key_ff[NBW_OUT-(4*gen_idx+1)*NBW_WORD-1-:NBW_WORD] ^ next_rkey[NBW_KEY-NBW_WORD-1-:NBW_WORD];
    next_rkey[NBW_KEY-3*NBW_WORD-1-:NBW_WORD] = expanded_key_ff[NBW_OUT-(4*gen_idx  )*NBW_WORD-1-:NBW_WORD] ^ next_rkey[NBW_KEY-2*NBW_WORD-1-:NBW_WORD];
end

always_comb begin : expanded_key_update
    expanded_key_nx = expanded_key_ff;

    if(i_start) begin
        expanded_key_nx = {i_key, {(NBW_OUT-NBW_KEY){1'b0}}};
    end else if(|key_exp_steps_ff[STEPS-1:0] && !key_exp_steps_ff[STEPS]) begin
        expanded_key_nx[NBW_OUT-(gen_idx+2)*NBW_KEY +: NBW_KEY] = next_rkey;
    end
end

endmodule : aes128_key_expansion
