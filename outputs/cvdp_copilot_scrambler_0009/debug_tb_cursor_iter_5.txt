## Rationale
- The failure moved from 140 ns (`WAIT_CYCLES=4`) to 240 ns (`WAIT_CYCLES=14`), so the prior `DELAY_TAP` special-case for `WAIT==4` does not generalize and the next fix needs timing evidence for large `WAIT` values.
- The symptom is unchanged (`0x0 != 0xf7`): the model expects data while the DUT still drives zero, which points to `delay_done` / `enable_output` never asserting at the right cycle — not a compile or elaboration issue.
- With `WAIT=14`, `DELAY_WIDTH=18` and the current tap `DELAY_WIDTH-1` (bit 17) enables output ~2 cycles after the model (loop ~22 vs ~20), matching a late/zero DUT at 240 ns.
- Prior cocotb-aligned iverilog runs were decisive for `WAIT=4`; the same stimulus style (inputs before posedge, reset sequence) should be reused with `WAIT_CYCLES=14` and a check at 240 ns.
- Single-module design, no UVM/DPI — a parameterized TB is enough to compare `start_intra_ff[WAIT_CYCLES]` vs `start_intra_ff[DELAY_WIDTH-1]` on the failing configuration.

```verilog
`timescale 1ns/1ps

module tb_debug;
  localparam ROW_COL_WIDTH  = 16;
  localparam SUB_BLOCKS     = 4;
  localparam DATA_WIDTH     = ROW_COL_WIDTH * ROW_COL_WIDTH;
  localparam OUT_DATA_WIDTH = 8;
  localparam WAIT_CYCLES    = 14;          // failing harness configuration
  localparam DELAY_WIDTH    = WAIT_CYCLES + SUB_BLOCKS;
  localparam DELAY_TAP_CUR  = (WAIT_CYCLES == 4) ? WAIT_CYCLES : (DELAY_WIDTH - 1);
  localparam FAIL_TIME_NS   = 240;
  localparam CLK_HALF       = 5;

  reg clk;
  reg rst_n;
  reg i_valid;
  reg [DATA_WIDTH-1:0] in_data;
  wire [OUT_DATA_WIDTH-1:0] out_data;

  integer loop_idx;
  reg tb_mismatch;
  time mismatch_time;

  wire tap_wait_bit      = dut.start_intra_ff[WAIT_CYCLES];
  wire tap_delay_msb_bit = dut.start_intra_ff[DELAY_WIDTH - 1];

  deinter_block #(
    .ROW_COL_WIDTH (ROW_COL_WIDTH),
    .SUB_BLOCKS    (SUB_BLOCKS),
    .DATA_WIDTH    (DATA_WIDTH),
    .OUT_DATA_WIDTH(OUT_DATA_WIDTH),
    .WAIT_CYCLES   (WAIT_CYCLES)
  ) dut (
    .clk     (clk),
    .rst_n   (rst_n),
    .i_valid (i_valid),
    .in_data (in_data),
    .out_data(out_data)
  );

  initial clk = 1'b0;
  always #(CLK_HALF) clk = ~clk;

  initial begin
    $dumpfile("wave.vcd");
    $dumpvars(0, tb_debug);
  end

  task apply_inputs;
    input integer idx;
    begin
      if (idx < 4) begin
        i_valid = 1'b1;
        in_data = {DATA_WIDTH{1'b1}} ^ (idx * 256'h100);
      end else begin
        i_valid = 1'b0;
        in_data = {DATA_WIDTH{1'b0}};
      end
    end
  endtask

  task sample_and_check;
    input integer idx;
    reg expect_model_active;
    begin
      // Model first expects output around loop_i=20 for WAIT=14 (240ns)
      expect_model_active = (idx >= 20);

      $display("t=%0t loop_i=%0d out_data=0x%0h enable=%b delay_done=%b ff=0x%0h",
               $time, idx, out_data, dut.enable_output, dut.delay_done, dut.start_intra_ff);
      $display("  tap_cur=%0d=%b tap_wait=%0d=%b tap_msb=%0d=%b active=%b cnt=%0d",
               DELAY_TAP_CUR, dut.start_intra_ff[DELAY_TAP_CUR],
               WAIT_CYCLES, tap_wait_bit,
               DELAY_WIDTH-1, tap_delay_msb_bit,
               dut.output_active, dut.counter_output);

      if (expect_model_active && out_data == {OUT_DATA_WIDTH{1'b0}}) begin
        tb_mismatch   = 1'b1;
        mismatch_time = $time;
        $display("First mismatch occurred at time %0t", $time);
        $display("  loop_i=%0d model expects non-zero, DUT=0", idx);
      end

      if ($time == FAIL_TIME_NS && out_data == {OUT_DATA_WIDTH{1'b0}}) begin
        tb_mismatch   = 1'b1;
        mismatch_time = $time;
        $display("First mismatch occurred at time %0t", $time);
        $display("  240ns harness failure: enable=%b delay_done=%b ff=0x%0h",
                 dut.enable_output, dut.delay_done, dut.start_intra_ff);
      end
    end
  endtask

  initial begin
    tb_mismatch   = 1'b0;
    mismatch_time = 0;
    rst_n         = 1'b0;
    i_valid       = 1'b0;
    in_data       = {DATA_WIDTH{1'b0}};

    #10;
    rst_n = 1'b1;
    #10;
    @(posedge clk);

    for (loop_idx = 0; loop_idx < 25; loop_idx = loop_idx + 1) begin
      apply_inputs(loop_idx);
      @(posedge clk);
      sample_and_check(loop_idx);
    end

    #50;
    if (!tb_mismatch)
      $display("PASS: no zero-output mismatch in monitored window.");
    else
      $display("FAIL: tb_mismatch=1 first at time %0t", mismatch_time);
    $finish;
  end
endmodule
```
