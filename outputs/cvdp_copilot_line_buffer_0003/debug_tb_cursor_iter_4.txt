## Rationale
- Harness failures **persist after the snapshot fix** (e.g. `0x61d8aa0fa234… != 0xfa2340…` at 170 ns), so the next iteration needs **internal signal visibility**, not just another output hex diff.
- The cocotb loop applies **`update_inputs` before `add_line`** when both `i_valid` and `i_update_window` are 1; waves on `image_buffer_ff`, `image_buffer_snapshot`, and `buffer_view` at the posedge are the fastest way to confirm whether the mux selects the pre-shift buffer correctly.
- Some mismatches show **partial bus agreement** (`0xa23475 != 0xa23475d8aa3d`), which points to **hold vs live-update** behavior (`i_update_window`, `image_window_ff`) rather than a single wrong pixel index — that is a multi-cycle timing question VCD answers directly.
- The design is a **single iverilog-friendly module**; a deterministic mini-TB can replay the spec 10-cycle table plus an explicit **both-high stress cycle** without cocotb, `harness_library`, or random seed replay.
- The prior local TB failed on **`function automatic` syntax**, not on concept; an iverilog-2001-safe rewrite is low cost and targets the exact suspected failure class.

```verilog
`timescale 1ns/1ps

// Build:
//   iverilog -g2012 -o sim rtl/line_buffer.sv tb_debug.sv && vvp sim
// View:
//   gtkwave wave.vcd

module tb_debug;

    // Match first harness failure configuration
    localparam NBW_DATA  = 8;
    localparam NS_ROW    = 10;
    localparam NS_COLUMN = 8;
    localparam NS_R_OUT  = 4;
    localparam NS_C_OUT  = 3;
    localparam CONSTANT  = 255;
    localparam NBW_ROW   = 4;
    localparam NBW_COL   = 3;
    localparam WIN_W     = NBW_DATA * NS_R_OUT * NS_C_OUT; // 96

    reg clk;
    reg rst_async_n;
    reg [2:0]  i_mode;
    reg        i_valid;
    reg        i_update_window;
    reg [NBW_DATA*NS_COLUMN-1:0] i_row_image;
    reg [NBW_ROW-1:0] i_image_row_start;
    reg [NBW_COL-1:0] i_image_col_start;
    wire [WIN_W-1:0] o_image_window;

    reg        tb_mismatch;
    reg        tb_first_mismatch_done;
    integer    tb_cycle;

    // Reference line buffer for cocotb-order model (update before add_line)
    reg [NBW_DATA-1:0] ref_buf [0:9][0:7];
    reg [WIN_W-1:0]    ref_window;
    reg [WIN_W-1:0]    last_good_window;

    line_buffer #(
        .NBW_DATA (NBW_DATA),
        .NS_ROW   (NS_ROW),
        .NS_COLUMN(NS_COLUMN),
        .NBW_ROW  (NBW_ROW),
        .NBW_COL  (NBW_COL),
        .NBW_MODE (3),
        .NS_R_OUT (NS_R_OUT),
        .NS_C_OUT (NS_C_OUT),
        .CONSTANT (CONSTANT)
    ) dut (
        .clk               (clk),
        .rst_async_n       (rst_async_n),
        .i_mode            (i_mode),
        .i_valid           (i_valid),
        .i_update_window   (i_update_window),
        .i_row_image       (i_row_image),
        .i_image_row_start (i_image_row_start),
        .i_image_col_start (i_image_col_start),
        .o_image_window    (o_image_window)
    );

    initial clk = 1'b0;
    always #5 clk = ~clk;

    // --- reference helpers (iverilog-safe: no automatic functions) ---

    task ref_reset;
        integer r, c;
        begin
            for (r = 0; r < NS_ROW; r = r + 1)
                for (c = 0; c < NS_COLUMN; c = c + 1)
                    ref_buf[r][c] = {NBW_DATA{1'b0}};
            ref_window      = {WIN_W{1'b0}};
            last_good_window = {WIN_W{1'b0}};
        end
    endtask

    task ref_unpack_row;
        input [NBW_DATA*NS_COLUMN-1:0] row_word;
        integer c;
        reg [NBW_DATA-1:0] row_image [0:7];
        begin
            for (c = 0; c < NS_COLUMN; c = c + 1)
                row_image[NS_COLUMN-c-1] =
                    row_word[(c+1)*NBW_DATA-1 -: NBW_DATA];
            for (c = 0; c < NS_COLUMN; c = c + 1)
                ref_buf[0][c] = row_image[c];
            for (r = 1; r < NS_ROW; r = r + 1)
                for (c = 0; c < NS_COLUMN; c = c + 1)
                    ref_buf[r][c] = ref_buf[r-1][c];
        end
    endtask

    task ref_compute_window;
        input [2:0] mode;
        input [NBW_ROW-1:0] rs;
        input [NBW_COL-1:0] cs;
        integer wr, wc, sr, sc, row_dist, col_dist;
        reg [NBW_DATA-1:0] pix;
        reg [WIN_W-1:0] packed;
        begin
            packed = {WIN_W{1'b0}};
            for (wr = 0; wr < NS_R_OUT; wr = wr + 1) begin
                for (wc = 0; wc < NS_C_OUT; wc = wc + 1) begin
                    sr = rs + wr;
                    sc = cs + wc;
                    case (mode)
                        3'd0: pix = (sr >= NS_ROW || sc >= NS_COLUMN) ? {NBW_DATA{1'b0}} : ref_buf[sr][sc];
                        3'd1: pix = (sr >= NS_ROW || sc >= NS_COLUMN) ? CONSTANT[NBW_DATA-1:0] : ref_buf[sr][sc];
                        3'd2: begin
                            if (sr >= NS_ROW) sr = NS_ROW - 1;
                            if (sc >= NS_COLUMN) sc = NS_COLUMN - 1;
                            pix = ref_buf[sr][sc];
                        end
                        3'd3: begin
                            if (sr >= NS_ROW) begin
                                row_dist = sr - (NS_ROW - 1);
                                if (row_dist < (NS_ROW - 1))
                                    sr = (NS_ROW - 1) - row_dist;
                                else
                                    sr = 2 * (NS_ROW - 1) - row_dist;
                            end
                            if (sc >= NS_COLUMN) begin
                                col_dist = sc - (NS_COLUMN - 1);
                                if (col_dist < (NS_COLUMN - 1))
                                    sc = (NS_COLUMN - 1) - col_dist;
                                else
                                    sc = 2 * (NS_COLUMN - 1) - col_dist;
                            end
                            pix = ref_buf[sr][sc];
                        end
                        3'd4: begin
                            if (sr >= NS_ROW) sr = sr - NS_ROW;
                            if (sc >= NS_COLUMN) sc = sc - NS_COLUMN;
                            pix = ref_buf[sr][sc];
                        end
                        default: pix = {NBW_DATA{1'b0}};
                    endcase
                    packed[(wr*NS_C_OUT+wc+1)*NBW_DATA-1 -: NBW_DATA] = pix;
                end
            end
            ref_window = packed;
        end
    endtask

    // Drive inputs stable, posedge, cocotb model order, compare
    task drive_and_check;
        input [255:0] tag;
        input [2:0]   mode;
        input         valid;
        input         upd;
        input [63:0]  row_img;
        input [3:0]   rs;
        input [2:0]   cs;
        begin
            i_mode            = mode;
            i_valid           = valid;
            i_update_window   = upd;
            i_row_image       = row_img[NBW_DATA*NS_COLUMN-1:0];
            i_image_row_start = rs[NBW_ROW-1:0];
            i_image_col_start = cs[NBW_COL-1:0];

            @(posedge clk);
            #1;

            // cocotb order AFTER posedge
            if (upd)
                ref_compute_window(mode, rs[NBW_ROW-1:0], cs[NBW_COL-1:0]);
            if (valid)
                ref_unpack_row(row_img[NBW_DATA*NS_COLUMN-1:0]);

            if (upd)
                last_good_window = ref_window;

            if (o_image_window !== ref_window) begin
                tb_mismatch = 1'b1;
                if (!tb_first_mismatch_done) begin
                    tb_first_mismatch_done = 1'b1;
                    $display("First mismatch occurred at time %0t", $time);
                end
                $display("MISMATCH %0s cycle %0d @ %0t", tag, tb_cycle, $time);
                $display("  mode=%0d valid=%0d upd=%0d row=0x%h rs=%0d cs=%0d",
                         mode, valid, upd, row_img, rs, cs);
                $display("  DUT o_image_window = 0x%h", o_image_window);
                $display("  REF (cocotb order) = 0x%h", ref_window);
                $display("  dut.i_valid=%b dut.i_update_window=%b", dut.i_valid, dut.i_update_window);
            end else begin
                $display("PASS %0s cycle %0d @ %0t o=0x%h", tag, tb_cycle, $time, o_image_window);
            end

            tb_cycle = tb_cycle + 1;
        end
    endtask

    // Hold check: with upd=0 output must not change even if inputs move
    task check_hold;
        input [255:0] tag;
        begin
            @(posedge clk);
            #1;
            if (o_image_window !== last_good_window) begin
                tb_mismatch = 1'b1;
                if (!tb_first_mismatch_done) begin
                    tb_first_mismatch_done = 1'b1;
                    $display("First mismatch occurred at time %0t", $time);
                end
                $display("HOLD MISMATCH %0s @ %0t", tag, $time);
                $display("  expected hold = 0x%h got = 0x%h", last_good_window, o_image_window);
            end else begin
                $display("HOLD PASS %0s @ %0t", tag, $time);
            end
            tb_cycle = tb_cycle + 1;
        end
    endtask

    integer r, c;

    initial begin
        $dumpfile("wave.vcd");
        $dumpvars(0, tb_debug);

        tb_mismatch            = 1'b0;
        tb_first_mismatch_done = 1'b0;
        tb_cycle               = 0;

        clk = 1'b0;
        rst_async_n = 1'b0;
        i_mode = 3'd0;
        i_valid = 1'b0;
        i_update_window = 1'b0;
        i_row_image = {NBW_DATA*NS_COLUMN{1'b0}};
        i_image_row_start = {NBW_ROW{1'b0}};
        i_image_col_start = {NBW_COL{1'b0}};

        ref_reset();

        // Reset
        repeat (2) @(posedge clk);
        rst_async_n = 1'b1;
        @(posedge clk);
        #1;
        ref_reset();

        $display("=== TEST A: both-high ordering (10x8 harness geometry) ===");

        // Fill a few rows (valid only)
        drive_and_check("load0", 3'd4, 1'b1, 1'b0, 64'h0102030405060708, 4'd0, 3'd0);
        drive_and_check("load1", 3'd4, 1'b1, 1'b0, 64'h1112131415161718, 4'd0, 3'd0);
        drive_and_check("load2", 3'd4, 1'b1, 1'b0, 64'h2122232425262728, 4'd0, 3'd0);

        // Critical: valid AND update on same edge — window must use PRE-shift buffer
        drive_and_check("both_high", 3'd1, 1'b1, 1'b1, 64'ha0a1a2a3a4a5a6a7a8, 4'd1, 3'd2);

        // Hold: upd=0, change mode/start — output must stay frozen
        i_mode = 3'd0;
        i_valid = 1'b0;
        i_update_window = 1'b0;
        i_image_row_start = 4'd7;
        i_image_col_start = 3'd7;
        check_hold("after_both_high");

        $display("=== TEST B: spec 3x3 table (subset) with 2x2 window ===");
        // Re-instantiation not possible; use separate params note in waves for spec case.
        // Run reduced geometry by reusing same DUT is wrong size — document as optional second compile.
        // Instead, include one more both-high on current DUT with invalid mode (expect zero window)
        drive_and_check("invalid_mode_both", 3'd5, 1'b1, 1'b1, 64'hffffffffffffffff, 4'd3, 3'd3);

        if (tb_mismatch)
            $display("SUMMARY: FAIL — inspect wave.vcd:");
            $display("  dut.image_buffer_ff vs dut.image_buffer_snapshot vs dut.buffer_view");
            $display("  posedge when i_valid && i_update_window (both_high cycle)");
        else
            $display("SUMMARY: PASS");

        #20 $finish;
    end

endmodule
```

**Waveform checklist at the `both_high` posedge (time of first likely mismatch):**

| Signal | What to verify |
|--------|----------------|
| `dut.i_valid` & `dut.i_update_window` | Both high for the stress cycle |
| `dut.image_buffer_ff` | Updates to shifted row **after** posedge |
| `dut.image_buffer_snapshot` | Captures **pre-shift** contents when `i_valid` rises |
| `dut.buffer_view` | Must equal snapshot (not `image_buffer_ff`) while both inputs are high |
| `dut.window` / `dut.o_image_window` | Must match `REF` pre-shift window, not post-shift pixels from `0xa0…` row |
| Hold cycle | With `i_update_window=0`, `o_image_window` must equal `last_good_window` |
