## Rationale
- **Line 18 — `start` never captured:** `FF00` drives `control1` from `1'b1` instead of `start`. The pipeline control chain runs continuously after reset, so `done` asserts too early. The harness reports `Cycle 2: Expected latency 4, got 2`.
- **Lines 32–35 — wrong byte forwarding in stage 2:** `A21`/`B21` (ADD3 inputs) are fed from `A3`/`B3` instead of `A2`/`B2`, and `A31`/`B31` (stage-4 MSB staging) are fed from `A2`/`B2` instead of `A3`/`B3`. This misroutes upper operand bytes and corrupts partial sums (e.g. got `0x9CD2A56B` vs expected `0xD29E93A5`).
- **Lines 50–51 — swapped low sum bytes:** `S[7:0]` is loaded from `s13` (byte-1 sum) and `S[15:8]` from `s23` (byte-0 sum); they should be swapped so `S[7:0] ← s23` and `S[15:8] ← s13`.
- **Lines 83–94 — CLA carry is sequential, not combinational:** Carry bits are updated in separate clocked `always` blocks, so carries ripple over multiple cycles inside one adder stage. The spec requires combinational carry lookahead with registered outputs only.
- **Line 104 — wrong carry-out tap:** `carry1 = C[WIDTH-1]` uses the carry *into* the MSB bit; the final carry-out is `C[WIDTH]`. This explains cases where expected `Co=1` but got `Co=0` (e.g. `0x8484D609 + 0xB1F05663`).
- **Control chain (lines 18, 29, 39, 48) — correct once `start` is wired:** `control1 → control2 → control3 → done` gives the required 4-cycle latency from `start` assertion to `done`.

module carry_lookahead_adder #(parameter WIDTH = 32) (
    input [WIDTH-1:0] A,    // WIDTH bit operand for addition
    input [WIDTH-1:0] B,    // WIDTH bit operand for addition
    input Cin, clk, reset,  // Cin -> carry in, clk -> Synchronous to posedge of clk , reset -> Synchronous active high reset           
    output reg [WIDTH-1:0] S, // Final sum or result 
    output reg carry          // Final carry out           
);

    wire [WIDTH-1:0] P;
    wire [WIDTH-1:0] G;
    wire [WIDTH:0] C;

    wire [WIDTH-1:0] S1;
    wire carry1;

    // Generate (G) and Propagate (P) 
    genvar i;
    generate
        for (i = 0; i < WIDTH; i = i + 1) begin
            assign P[i] = A[i] ^ B[i];   
            assign G[i] = A[i] & B[i];   
        end
    endgenerate

    // Combinational carry lookahead
    assign C[0] = Cin;
    generate
        for (i = 0; i < WIDTH; i = i + 1) begin
            assign C[i+1] = G[i] | (P[i] & C[i]);
        end
    endgenerate

    // Final sum
    generate
        for (i = 0; i < WIDTH; i = i + 1) begin
            assign S1[i] = P[i] ^ C[i]; 
        end
    endgenerate

    assign carry1 = C[WIDTH];

    always @(posedge clk) begin
        if (reset) begin
            S <= 0; 
            carry <= 0;   
        end else begin
            S <= S1; 
            carry <= carry1;
        end 
    end

endmodule 

module dff #(parameter WIDTH = 1) (
    input clk,               // clk -> Synchronous to posedge of clk 
    input reset,             // reset -> Synchronous active high reset
    input [WIDTH-1:0] D,     // D -> Data signal 
    output reg [WIDTH-1:0] Q // Q -> Output signal 
);
    always @(posedge clk) begin
        if (reset)
            Q <= {WIDTH{1'b0}};
        else
            Q <= D;
    end
endmodule
