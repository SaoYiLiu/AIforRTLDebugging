## Rationale
- **Harness failure:** After `data_out` latches to `00000100`, the test waits two more `src_clk` cycles and asserts `ack_out == 1`; the log shows `ack_out after 2 more src_clk = 0` (lines 41–50, 54–57).
- **Root cause (lines 46–50):** `ack` is asserted for exactly one `dst_clk` cycle, then forced low every other cycle via `else ack <= 1'b0`. That produces a single-cycle pulse in the destination domain.
- **CDC issue:** That pulse is synchronized into `src_clk` with the 2-FF `enable_synch_1` (line 54). When `dst_clk` is faster than `src_clk`, a one-cycle pulse can fall entirely between two `src_clk` posedges and never get sampled — matching the bug description.
- **Test timing:** The check runs after a `dst_clk` rising edge plus several `src_clk` edges, so the narrow `ack` pulse can already be low before the synchronizer propagates it, yielding `syncd_ack = 0` and `ack_out = 0`.
- **Fix:** Hold `ack` high after `anded_req` until the synchronized acknowledgment is seen back in the `dst_clk` domain (`syncd_ack_dst`), using a feedback synchronizer `ack_fb_synch`.
- **Clear path:** `syncd_ack_dst` only goes high after `syncd_ack` is stable in `src_clk`, so `ack` stays high long enough for reliable CDC, then clears for the next transfer.
- **`ack_out` (line 57):** Keep `assign ack_out = syncd_ack`; the test expects a level `1`, not a one-cycle edge pulse.

### rtl/mux_synch.sv
```verilog
module mux_synch (

input [7:0] data_in,   			//asynchronous data input
input req,                  		//indicating that data is available at the data_in input
input dst_clk,                 		//destination clock
input src_clk,                 		//source clock
input nrst,                    		//asynchronous reset 
output reg [7:0] data_out,              //synchronized version of data_in to the destination clock domain
output ack_out ); 		

wire syncd_req,anded_req,syncd_ack,syncd_ack_dst;
reg syncd_req_1,ack;

                        						
nff  req_synch_0 (.d_in(req),.dst_clk(dst_clk),.rst(nrst),.syncd(syncd_req)) ;		//2-flop synchronizer for the enable input

                                
always_ff @(posedge dst_clk)                     					//one clock cycle delayed synced_enable
begin
    syncd_req_1 <= syncd_req;
end

assign anded_req = (!syncd_req_1 && syncd_req);    					//posedge detector

	
always_ff @(posedge dst_clk or negedge nrst)
begin                                                   
	if(!nrst)
		data_out <= 1'b0;               					//forcing the output data_in to zero when an active-low asynchronous reset is detected.

	else if (anded_req==1'b1)
		data_out <= data_in;                    				//latching data_in to data_out when the enable signal is available.
	
	else
		data_out <= data_out;                   				//holds the data till next req comes.
end

// acknowledgment signal generation - hold until synchronized back to dst domain
always_ff@(posedge dst_clk or negedge nrst)
begin
	if(!nrst)
		ack <= 1'b0; 

	else if (anded_req==1'b1)
		ack <= 1'b1;
	
	else if (syncd_ack_dst==1'b1)
		ack <= 1'b0;
end

//changing the clock domain of the ack signal
nff  enable_synch_1 (.d_in(ack),.dst_clk(src_clk),.rst(nrst),.syncd(syncd_ack)) ;

// synchronize ack feedback to dst domain to safely deassert ack pulse
nff  ack_fb_synch (.d_in(syncd_ack),.dst_clk(dst_clk),.rst(nrst),.syncd(syncd_ack_dst)) ;

//edge detector circuit
assign ack_out = syncd_ack;

endmodule

module nff  (
	
	input d_in,   									//input data that needs to be synchronized to the dst_clk domain.
	input dst_clk,     								//destination domain clock.
	input rst,         								//asynchronous active-low reset
	output reg  syncd 								//synced output, which is a 2-clock-cycle delayed version of d_in.
	                   );

reg   dmeta;             								//register to hold output.

always@(posedge dst_clk or negedge rst)  
begin
	if(!rst)              								//active-low asynchronous reset
  begin
    syncd <= 1'b0;      								//resetting the synced register to 0
    dmeta <= 1'b0;      								//resetting dmeta register to 0
  end
  else
  begin
    dmeta <= d_in;      								//passing d_in to dmeta
    syncd <= dmeta;     								//passing dmeta to syncd
  end
end

endmodule
```
