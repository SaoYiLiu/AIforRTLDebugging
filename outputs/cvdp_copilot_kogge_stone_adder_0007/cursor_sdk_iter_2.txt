## Rationale
- **Lines 55–61 — broken span-1 prefix at bits 3 and 7:** The condition `i != 3 && i != 7` skips the normal Kogge-Stone merge for those positions, so carry does not propagate through bits 3 and 7. That matches harness off-by-small-amount failures (e.g. Test 89: expected `81517`, got `81533`) and the repeated-operand case where the same `A/B` yields `0x68AC`, `0x68AE`, and `0x68AB`.

- **Lines 68–71 — forced reset of bit 10 in stage 2:** Forcing `G2[10]=0` and `P2[10]=1` breaks the span-2 prefix at bit 10, corrupting carries above that bit. This aligns with large mismatches such as Test 87 (`62056` vs `129640`) and Test 94 (`72051` vs `4467`).

- **Lines 83–86 — swapped generate/propagate at bit 5:** Assigning `G3[5]=P2[5]` and `P3[5]=G2[5]` inverts prefix semantics at bit 5. That explains incrementing-operand errors like cycle 4 (`0x0004` expected vs `0x0002` actual) and random cases off by 1 (e.g. `0x4D03` vs `0x4D02`).

- **Line 105 — incorrect MSB sum:** `sum_comb[16] = carry[16] ^ carry[5]` XORs the final carry with an intermediate carry, corrupting the 17th result bit. Special cases with carry-out fail accordingly (e.g. `A=255, B=65280`: expected `65535`, got `65599`; decrementing `0xFFFF+0x0001` expected `0x10000` but got wrong carry behavior).

- **Missing span-8 prefix stage (architectural gap at lines 96–106):** A 16-bit Kogge-Stone adder needs four prefix distances (1, 2, 4, 8). The original RTL only computes through `G3/P3` (span 4) before forming carries, so bits 8–15 never see full carry propagation from bit 0. Adding `G4/P4` at `stage >= 3` and using them in the carry chain completes the algorithm.

- **Lines 53–94 — `stage`-gated prefix is otherwise structurally sound:** The FSM (stages 0→4, latch `Sum` at `stage==4`) is fine once prefix math is corrected; inconsistent repeated results were a symptom of corrupted prefix terms, not missing registers.

module kogge_stone_adder (
    input logic clk,
    input logic reset,
    input logic [15:0] A,
    input logic [15:0] B,
    input logic start,
    output logic [16:0] Sum,
    output logic done
);

    logic [15:0] G0, G1, G2, G3, G4;
    logic [15:0] P0, P1, P2, P3, P4;
    logic [16:0] carry;
    logic [16:0] sum_comb;
    logic [3:0] stage;
    logic active;

    always_ff @(posedge clk or posedge reset) begin
        if (reset) begin
            Sum <= 0;
            done <= 0;
            active <= 0;
            stage <= 0;
        end else if (start && !active) begin
            active <= 1;
            stage <= 0;
            done <= 0;
            Sum <= 0;
        end else if (active) begin
            if (stage == 4) begin
                Sum <= sum_comb;
                done <= 1;
                active <= 0;
            end else begin
                stage <= stage + 1;
            end
        end else if (!start) begin
            done <= 0;
        end
    end

    always_comb begin
        G1 = 0; G2 = 0; G3 = 0; G4 = 0;
        P1 = 0; P2 = 0; P3 = 0; P4 = 0;
        carry = 0;
        sum_comb = 0;

        for (int i = 0; i < 16; i++) begin
            G0[i] = A[i] & B[i];
            P0[i] = A[i] ^ B[i];
        end

        if (stage >= 0) begin
            for (int i = 0; i < 16; i++) begin
                if (i >= 1) begin
                    G1[i] = G0[i] | (P0[i] & G0[i - 1]);
                    P1[i] = P0[i] & P0[i - 1];
                end else begin
                    G1[i] = G0[i];
                    P1[i] = P0[i];
                end
            end
        end

        if (stage >= 1) begin
            for (int i = 0; i < 16; i++) begin
                if (i >= 2) begin
                    G2[i] = G1[i] | (P1[i] & G1[i - 2]);
                    P2[i] = P1[i] & P1[i - 2];
                end else begin
                    G2[i] = G1[i];
                    P2[i] = P1[i];
                end
            end
        end

        if (stage >= 2) begin
            for (int i = 0; i < 16; i++) begin
                if (i >= 4) begin
                    G3[i] = G2[i] | (P2[i] & G2[i - 4]);
                    P3[i] = P2[i] & P2[i - 4];
                end else begin
                    G3[i] = G2[i];
                    P3[i] = P2[i];
                end
            end
        end

        if (stage >= 3) begin
            for (int i = 0; i < 16; i++) begin
                if (i >= 8) begin
                    G4[i] = G3[i] | (P3[i] & G3[i - 8]);
                    P4[i] = P3[i] & P3[i - 8];
                end else begin
                    G4[i] = G3[i];
                    P4[i] = P3[i];
                end
            end

            carry[0] = 0;
            for (int i = 1; i <= 16; i++) begin
                carry[i] = G4[i - 1] | (P4[i - 1] & carry[i - 1]);
            end

            for (int i = 0; i < 16; i++) begin
                sum_comb[i] = P0[i] ^ carry[i];
            end
            sum_comb[16] = carry[16];
        end
    end
endmodule
